📦 1. Variabel & Tipe Data

📝 String (Teks)

nama = "Ari"
nama = 'Ari'

multi
baris"""

🔢 Integer (Angka Bulat)

umur = 17
negatif = -5
positif = 100

🟠 Float (Angka Desimal)

tinggi = 1.75
pi = 3.14159
nol_koma = 0.5

✅ Boolean (Benar/Salah)

aktif = True
salah = False

📋 2. List, Tuple, Dict, Set

📝 List (Daftar)

# Membuat list
nilai = [80, 90, 85, 70]
nama = ["Ari", "Budi", "Citra"]

# Mengakses
nilai[0]    # 80 (pertama)
nilai[-1]   # 70 (terakhir)
nilai[0:2]  # [80, 90]

# Menambah
nilai.append(95)
nilai.insert(0, 100)

📦 Tuple

# Tuple (tidak bisa diubah)
koordinat = (10, 20)
titik = (x, y, z)

# Mengakses
koordinat[0]  # 10
koordinat[1]  # 20

🔑 Dictionary (Dict)

# Key-Value pairs
siswa = {
    "nama": "Ari",
    "umur": 17,
    "kelas": "XII-IPA1"
}

# Mengakses
siswa["nama"]     # "Ari"
siswa.get("umur") # 17

# Menambah/Mengubah
siswa["nilai"] = 90

🎲 Set

# Kumpulan unik
angka = {1, 2, 3, 3, 3}
# {1, 2, 3}

# Operasi set
A = {1, 2, 3}
B = {3, 4, 5}

A | B  # Union: {1,2,3,4,5}
A & B  # Intersection: {3}
A - B  # Difference: {1,2}

➕ 3. Operator

🧮 Aritmatika

a, b = 10, 3

a + b   # 13  (tambah)
a - b   # 7   (kurang)
a * b   # 30  (kali)
a / b   # 3.33 (bagi)
a // b  # 3   (bagi bulat)
a % b   # 1   (sisa bagi)
a ** b  # 1000 (pangkat)

🔀 Perbandingan

x, y = 5, 10

x == y  # False (sama dengan)
x != y  # True  (tidak sama)
x > y   # False (lebih besar)
x < y   # True  (lebih kecil)
x >= y  # False (lebih besar/sama)
x <= y  # True  (lebih kecil/sama)

🔗 Logika

a, b = True, False

a and b  # False
a or b   # True
not a    # False

# Short-circuit
x and y  # xFalse, baru y
x or y   # xTrue, baru y

🔧 Assignment

x = 5
x += 3   # x = x + 3  -> 8
x -= 2   # x = x - 2  -> 6
x *= 2   # x = x * 2  -> 12
x /= 3   # x = x / 3  -> 4
x //= 2  # x = x // 2 -> 2
x %= 2   # x = x % 2  -> 0

🔄 4. Control Flow (If, For, While)

🔀 If-Else

umur = 17

if umur >= 18:
    print("Dewasa")
elif umur >= 13:
    print("Remaja")
else:
    print("Anak-anak")

🔁 For Loop

# Range
for i in range(5):
    print(i)  # 0,1,2,3,4

for i in range(1, 6):
    print(i)  # 1,2,3,4,5

# List
for nilai in [80, 90, 85]:
    print(nilai)

⏳ While Loop

i = 0
while i < 5:
    print(i)
    i += 1

# Dengan break
while True:
    x = input("Ketik 'exit': ")
    if x == "exit":
        break

⏭️ Break, Continue, Pass

for i in range(10):
    if i == 3:
        continue  # Skip ini
    if i == 7:
        break     # Stop loop
    print(i)

# Pass (placeholder)
if True:
    pass  # Nanti diisi

⚙️ 5. Functions

📌 Fungsi Dasar

def sapa(nama):
    return f"Halo, {nama}!"

# Panggil
print(sapa("Ari"))

# Default parameter
def greet(nama, pesan="Halo"):
    return f"{pesan}, {nama}!"

📊 Multiple Return

def kalkulasi(a, b):
    return a+b, a-b, a*b, a/b

tambah, kurang, kali, bagi = kalkulasi(10, 3)

⚡ Lambda (Anonim)

# Satu baris
kuadrat = lambda x: x ** 2
print(kuadrat(5))  # 25

# Dengan if
genap = lambda x: "Genap" if x % 2 == 0 else "Ganjil"

📦 *args & **kwargs

def total(*angka):
    return sum(angka)

print(total(1,2,3,4))  # 10

def info(**data):
    for key, value in data.items():
        print(f"{key}: {value}")

info(nama="Ari", kelas="XII")

📝 6. String Methods

✂️ Mengubah Case

text = "Hello World"

text.lower()        # "hello world"
text.upper()        # "HELLO WORLD"
text.title()        # "Hello World"
text.capitalize()   # "Hello world"

🔍 Mencari & Mengganti

text = "Hello World"

text.find("World")   # 6 (index)
text.replace("World", "Python")
                     # "Hello Python"

text.count("l")     # 3

✂️ Memotong & Split

text = "  Hello World  "

text.strip()      # "Hello World"
text.split()     # ["Hello", "World"]
text.split(",")  # pisah dg koma

"-".join(["a","b","c"])  # "a-b-c"

✅ Pengecekan

text = "Hello123"

text.isalpha()    # False (ada angka)
text.isdigit()    # False
text.isalnum()    # True (huruf/angka)
text.startswith("Hello")  # True
text.endswith("123")     # True

📂 7. File Handling

📖 Membaca File

# Cara aman (auto close)
with open("file.txt", "r") as f:
    content = f.read()

# Baris per baris
with open("file.txt", "r") as f:
    for line in f:
        print(line.strip())

✍️ Menulis File

# Tulis (hapus isi lama)
with open("file.txt", "w") as f:
    f.write("Halo dunia!")

# Tambah (append)
with open("file.txt", "a") as f:
    f.write("\nBaris baru")

📋 Mode File

"r"  # Read (baca)
"w"  # Write (tulis, hapus)
"a"  # Append (tambah)
"rb" # Read Binary
"wb" # Write Binary

⚠️ 8. Exception Handling

🛡️ Try-Except

try:
    result = 10 / 0
except ZeroDivisionError:
    print("Tidak boleh bagi nol!")
except Exception as e:
    print(f"Error: {e}")
finally:
    print("Selalu dijalankan")

❓ Raise Error

def bagi(a, b):
    if b == 0:
        raise ValueError("Pembagi nol!")
    return a / b

try:
    bagi(10, 0)
except ValueError as e:
    print(e)

🏗️ 9. OOP (Object-Oriented Programming)

📦 Class Dasar

class Siswa:
    def __init__(self, nama, umur):
        self.nama = nama
        self.umur =umur
    
    def sapa(self):
        return f"Halo, {self.nama}!"

# Buat object
siswa = Siswa("Ari", 17)
print(siswa.sapa())

🧬 Inheritance

class Manusia:
    def __init__(self, nama):
        self.nama = nama
    
    def makan(self):
        print("Makan...")

class Siswa(Manusia):
    def belajar(self):
        print("Belajar...")

siswa = Siswa("Ari")
siswa.makan()  # dari parent
siswa.belajar()

🔒 Encapsulation

class Akun:
    def __init__(self):
        self.__saldo = 1000  # Private
    
    def get_saldo(self):
        return self.__saldo
    
    def set_saldo(self, nilai):
        if nilai >= 0:
            self.__saldo = nilai

🔄 Polymorphism

class Kucing:
    def suara(self):
        return "Meong"

class Anjing:
    def suara(self):
        return "Guk-guk"

def cetak_suara(hewan):
    print(hewan.suara())

cetak_suara(Kucing())  # Meong
cetak_suara(Anjing()) # Guk-guk

💡 10. Tips & Tricks

🔄 List Comprehension

# Lama
kuadrat = []
for i in range(5):
    kuadrat.append(i**2)

# Baru (satu baris)
kuadrat = [i**2 for i in range(5)]

# Dengan kondisi
genap = [i for i in range(10) if i%2==0]

📊 Fungsi Bawaan

nilai = [80, 90, 85]

len(nilai)     # 3 (jumlah)
sum(nilai)     # 255 (total)
max(nilai)     # 90 (terbesar)
min(nilai)     # 80 (terkecil)
sorted(nilai) # [80,85,90]
reversed(nilai)

🎯 Enumerate & Zip

nama = ["Ari", "Budi"]

# Dengan index
for i, n in enumerate(nama):
    print(f"{i}: {n}")

# Gabung 2 list
for n, nilai in zip(nama, [80, 90]):
    print(f"{n}: {nilai}")

🎨 F-String

nama = "Ari"
umur = 17

# Cara lama
print("Nama: " + nama)

# F-string (recommended)
print(f"Nama: {nama}, Umur: {umur}")

# Format angka
print(f"Pi: {3.14159:.2f}")  # 3.14
